# Runtime dependencies for the congbobanan five-pipeline chain.
#
# Install for local development:
#     pip install -r packages/datasites/congbobanan/requirements.txt

# --- env + installer -----------------------------------------------
# uv is the recommended installer + lockfile tool for the monorepo
# (see docs/00-overview/repo-layout.md "Tooling"). Listing it here
# keeps `uv pip install -r requirements.txt` a one-step bootstrap
# on fresh hosts that only ship plain `pip`.
uv>=0.4

# --- pipeline runtime (NeMo Curator 2.x) ---------------------------
# ProcessingStage / Pipeline / DocumentBatch + XennaExecutor +
# RayActorPoolExecutor + RayDataExecutor all live here. Swap
# `text_cpu` for `text_cuda12` on a CUDA host or install the `[gpu]`
# workspace extra.
nemo-curator[text_cpu]>=0.7

# --- config + CLI --------------------------------------------------
omegaconf>=2.3

# --- HTTP + HTML ---------------------------------------------------
requests>=2.31
requests[socks]>=2.31      # SOCKS5 proxy support for VN-egress routing

# --- data I/O ------------------------------------------------------
pandas>=2.1
pyarrow>=15.0
pyyaml>=6.0

# --- parser (local PDF + NIM PDF rasterization) -------------------
# pypdf: native text extraction for digital PDFs. Used by the local
#        runtime and by the hybrid runtime's first pass.
# pypdfium2 + Pillow: page -> PNG raster for cfg.parser.runtime in
#            {nim, hybrid}. nemoretriever-parse accepts images only,
#            so the NIM path rasterizes every page via pypdfium2 and
#            encodes PNG via Pillow before upload. Pure-wheel, no
#            system-level poppler / mupdf dep.
pypdf>=4.0
pypdfium2>=4.24
Pillow>=10.0

# --- NIM / NVIDIA endpoints ---------------------------------------
openai>=1.30               # OpenAI-compatible NIM client

# --- reducer (CPU fallback) ---------------------------------------
scikit-learn>=1.3          # PCA / HDBSCAN / t-SNE fallbacks
# umap-learn 0.5.6 is the first release compatible with scikit-learn
# >=1.6 and ships wheels for numba 0.60+/0.65 (Python 3.12). Older
# pins resolve to 0.5.4 which fails to import against current sklearn.
umap-learn>=0.5.6
