# Runtime dependencies for the anle 6-stage pipeline.
#
# Install for local development:
#     pip install -r packages/datasites/anle/requirements.txt
#
# Optional GPU extras (rapids / cuml / cugraph) are kept commented out;
# uncomment on a CUDA host to enable GPU paths in the reducer (cuML
# UMAP/HDBSCAN). The CPU `umap-learn` fallback below is always required.

# --- pipeline runtime (NeMo Curator 2.x) ---------------------------
# ProcessingStage / Pipeline / DocumentBatch / the XennaExecutor +
# RayActorPoolExecutor + RayDataExecutor backends all live here.
# `text_cpu` pulls parquet/jsonl readers+writers and the download
# composite (DocumentDownloadExtractStage). Replace with `text_cuda12`
# on a CUDA host or install the `[gpu]` workspace extra.
nemo-curator[text_cpu]>=0.7

# --- env + installer ------------------------------------------------
# uv is the recommended installer + lockfile tool for the monorepo
# (see docs/00-overview/repo-layout.md "Tooling"). Listing it here
# keeps `uv pip install -r requirements.txt` a one-step bootstrap
# on fresh hosts that only ship plain `pip`.
uv>=0.4

# --- config + CLI ---------------------------------------------------
omegaconf>=2.3

# --- HTTP + HTML ----------------------------------------------------
requests>=2.31
requests[socks]>=2.31      # SOCKS5 proxy support for Vietnam-geo hosts
beautifulsoup4>=4.12

# --- data I/O -------------------------------------------------------
pandas>=2.1
pyarrow>=15.0
pyyaml>=6.0

# --- parser (local PDF + NIM PDF rasterization) --------------------
# pypdf: native text extraction for digital PDFs. Used by the local
#        runtime and by the hybrid runtime's first pass.
# pypdfium2 + Pillow: page -> PNG raster for cfg.parser.runtime in
#            {nim, hybrid}. nemoretriever-parse accepts images only,
#            so the NIM path rasterizes every page via pypdfium2 and
#            encodes PNG via Pillow before upload. Pure-wheel, no
#            system-level poppler / mupdf dep.
pypdf>=4.0
pypdfium2>=4.24
Pillow>=10.0

# --- NIM / NVIDIA endpoints ----------------------------------------
openai>=1.30               # OpenAI-compatible NIM client

# --- HF embedder (optional; required only if cfg.embedder.runtime=hf)
# Curator's EmbeddingCreatorStage uses sentence-transformers; install
# only on hosts where the HF runtime is actually enabled.
# transformers>=4.43
# torch>=2.2
# sentence-transformers>=3.0

# --- reducer (CPU fallback) ----------------------------------------
scikit-learn>=1.3          # PCA / HDBSCAN / t-SNE fallbacks
# umap-learn 0.5.6 is the first release compatible with scikit-learn
# >=1.6 and ships wheels for numba 0.60+/0.65 (Python 3.12). Older
# pins resolve to 0.5.4 which fails to import against current sklearn.
umap-learn>=0.5.6

# --- visualizer (out of pipeline; consumes parquet under apps/visualizer)
plotly>=5.20
kaleido>=0.2               # static image export (PNG)
networkx>=3.2
nbformat>=5.9

# --- optional GPU acceleration -------------------------------------
# Uncomment on a CUDA host; requires the NVIDIA rapids conda channel

# cudf-cu13
# cuml-cu13
# cugraph-cu13
